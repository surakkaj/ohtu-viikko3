18      public boolean logIn(String username, String password) {
19          for (User user : userDao.listAll()) {
20              if (user.getUsername().equals(username)
21                      && user.getPassword().equals(password)) {
22                  return true;
23              }
24          }
25  
26          return false;
27      }

29      public boolean createUser(String username, String password) {
30          if (userDao.findByName(username) != null) {
31              return false;
32          }
33  
34          if (invalid(username, password)) {
35              System.out.println("invalid username or password");
36              return false;
37          }
38  
39          userDao.add(new User(username, password));
40  
41          return true;
42      }
